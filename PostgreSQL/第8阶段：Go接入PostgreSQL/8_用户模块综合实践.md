# 8. 用户模块综合实践

本节目标：用 Go + pgxpool 写一个完整用户模块，包含创建用户、根据 ID 查询、根据邮箱查询、更新用户、删除用户，并在邮箱重复时返回明确错误。

本节会把前面的知识串起来：

- `pgxpool`
- 参数化查询
- `QueryRow`
- `Query`
- `Exec`
- `RETURNING`
- `RowsAffected`
- `pgx.ErrNoRows`
- `pgconn.PgError`

## 一、数据库表

```sql
CREATE SCHEMA IF NOT EXISTS app;

DROP TABLE IF EXISTS app.users;

CREATE TABLE app.users (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    email text NOT NULL,
    username text NOT NULL,
    display_name text,
    bio text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    deleted_at timestamptz,
    CONSTRAINT users_email_unique UNIQUE (email),
    CONSTRAINT users_email_not_empty_check CHECK (length(trim(email)) > 0),
    CONSTRAINT users_username_not_empty_check CHECK (length(trim(username)) > 0)
);

CREATE INDEX users_deleted_at_idx
ON app.users (deleted_at);
```

## 三、db.go

```go
package db

import (
	"context"
	"time"

	"github.com/jackc/pgx/v5/pgxpool"
)

func NewPool(ctx context.Context, databaseURL string) (*pgxpool.Pool, error) {
	config, err := pgxpool.ParseConfig(databaseURL)
	if err != nil {
		return nil, err
	}

	config.MaxConns = 10
	config.MinConns = 2
	config.MaxConnLifetime = time.Hour
	config.MaxConnIdleTime = 30 * time.Minute
	config.HealthCheckPeriod = time.Minute

	pool, err := pgxpool.NewWithConfig(ctx, config)
	if err != nil {
		return nil, err
	}

	if err := pool.Ping(ctx); err != nil {
		pool.Close()
		return nil, err
	}

	return pool, nil
}
```

## 五、errors.go

```go
package user

import (
	"errors"

	"github.com/jackc/pgx/v5/pgconn"
)

var (
	ErrUserNotFound     = errors.New("user not found")
	ErrEmailAlreadyUsed = errors.New("email already used")
	ErrInvalidUserInput = errors.New("invalid user input")
)

const (
	pgErrUniqueViolation  = "23505"
	pgErrNotNullViolation = "23502"
	pgErrCheckViolation   = "23514"
)

func mapDBError(err error) error {
	if err == nil {
		return nil
	}

	var pgErr *pgconn.PgError
	if !errors.As(err, &pgErr) {
		return err
	}

	switch pgErr.Code {
	case pgErrUniqueViolation:
		if pgErr.ConstraintName == "users_email_unique" {
			return ErrEmailAlreadyUsed
		}
		return ErrInvalidUserInput
	case pgErrNotNullViolation, pgErrCheckViolation:
		return ErrInvalidUserInput
	default:
		return err
	}
}
```

## 七、Create：创建用户

```go
func (r *Repository) Create(ctx context.Context, p CreateUserParams) (*User, error) {
	var u User

	err := r.pool.QueryRow(ctx, `
		INSERT INTO app.users (email, username, display_name, bio)
		VALUES ($1, $2, $3, $4)
		RETURNING id, email, username, display_name, bio, created_at, updated_at
	`, p.Email, p.Username, p.DisplayName, p.Bio).Scan(
		&u.ID,
		&u.Email,
		&u.Username,
		&u.DisplayName,
		&u.Bio,
		&u.CreatedAt,
		&u.UpdatedAt,
	)
	if err != nil {
		return nil, mapDBError(err)
	}

	return &u, nil
}
```

func (r *Repository) GetByID(ctx context.Context, id int64) (*User, error) {
	var u User

	err := r.pool.QueryRow(ctx, `
		SELECT id, email, username, display_name, bio, created_at, updated_at
		FROM app.users
		WHERE id = $1
		  AND deleted_at IS NULL
	`, id).Scan(
		&u.ID,
		&u.Email,
		&u.Username,
		&u.DisplayName,
		&u.Bio,
		&u.CreatedAt,
		&u.UpdatedAt,
	)
	if err != nil {
		if errors.Is(err, pgx.ErrNoRows) {
			return nil, ErrUserNotFound
		}
		return nil, err
	}

	return &u, nil
}

## 十、List：查询用户列表

```go
func (r *Repository) List(ctx context.Context, limit int32, offset int32) ([]User, error) {
	rows, err := r.pool.Query(ctx, `
		SELECT id, email, username, display_name, bio, created_at, updated_at
		FROM app.users
		WHERE deleted_at IS NULL
		ORDER BY id
		LIMIT $1 OFFSET $2
	`, limit, offset)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	users := make([]User, 0)
	for rows.Next() {
		var u User
		if err := rows.Scan(
			&u.ID,
			&u.Email,
			&u.Username,
			&u.DisplayName,
			&u.Bio,
			&u.CreatedAt,
			&u.UpdatedAt,
		); err != nil {
			return nil, err
		}
		users = append(users, u)
	}

	if err := rows.Err(); err != nil {
		return nil, err
	}

	return users, nil
}
```

func (r *Repository) Update(ctx context.Context, p UpdateUserParams) (*User, error) {
	var u User

	err := r.pool.QueryRow(ctx, `
		UPDATE app.users
		SET
			username = $2,
			display_name = $3,
			bio = $4,
			updated_at = now()
		WHERE id = $1
		  AND deleted_at IS NULL

## 十二、Delete：软删除用户

```go
func (r *Repository) Delete(ctx context.Context, id int64) error {
	tag, err := r.pool.Exec(ctx, `
		UPDATE app.users
		SET
			deleted_at = now(),
			updated_at = now()
		WHERE id = $1
		  AND deleted_at IS NULL
	`, id)
	if err != nil {
		return err
	}

	if tag.RowsAffected() == 0 {
		return ErrUserNotFound
	}

	return nil
}
```

## 十三、main.go 示例

```go
package main

import (
	"context"
	"errors"
	"fmt"
	"log"
	"os"
	"time"

	"go-pg-demo/internal/db"
	"go-pg-demo/internal/user"
)

func main() {
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	databaseURL := os.Getenv("DATABASE_URL")
	if databaseURL == "" {
		log.Fatal("DATABASE_URL is empty")
	}

	pool, err := db.NewPool(ctx, databaseURL)
	if err != nil {
		log.Fatalf("connect database: %v", err)
	}
	defer pool.Close()

	users := user.NewRepository(pool)

	created, err := users.Create(ctx, user.CreateUserParams{
		Email:    "alice@example.com",
		Username: "alice",
	})
	if err != nil {
		if errors.Is(err, user.ErrEmailAlreadyUsed) {
			log.Fatal("email already used")
		}
		log.Fatalf("create user: %v", err)
	}

	fmt.Printf("created user: %+v\n", created)

	got, err := users.GetByEmail(ctx, "alice@example.com")
	if err != nil {
		log.Fatalf("get user: %v", err)
	}

	fmt.Printf("found user: %+v\n", got)
}
```

## 十六、常见误区

### 1. Repository 里可以返回 HTTP 状态码吗？

不建议。

Repository 返回业务错误，HTTP handler 决定状态码。

### 2. SQL 分散在 handler 里可以吗？

小 demo 可以，真实项目不建议。

把 SQL 放在 Repository 更容易维护和测试。

### 3. 邮箱重复应该先查再插入吗？

不要只依赖先查。

并发下仍然可能重复。数据库唯一约束才是最终保障。

### 4. 删除用户一定要物理删除吗？

不一定。

这里用软删除是为了保留数据，也方便展示 `RowsAffected`。
