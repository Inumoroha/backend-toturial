# 9. 任务列表与详情

本节目标：实现 `GET /api/v1/tasks` 和 `GET /api/v1/tasks/:id`，支持分页、状态筛选和用户数据隔离。

---

## 一、列表接口设计

请求：

```http
GET /api/v1/tasks?page=1&page_size=10&status=todo
Authorization: Bearer token
```

响应：

```json
{
  "code": 0,
  "message": "ok",
  "data": {
    "items": [],
    "total": 0,
    "page": 1,
    "page_size": 10
  }
}
```

## 二、Repository 查询

```go
func (r *TaskRepository) ListByUserID(userID uint, page, pageSize int, status string) ([]model.Task, int64, error) {
    var tasks []model.Task
    var total int64

    query := r.db.Model(&model.Task{}).Where("user_id = ?", userID)
    if status != "" {
        query = query.Where("status = ?", status)
    }

    if err := query.Count(&total).Error; err != nil {
        return nil, 0, err
    }

    offset := (page - 1) * pageSize
    err := query.Order("created_at DESC").Offset(offset).Limit(pageSize).Find(&tasks).Error
    if err != nil {
        return nil, 0, err
    }

    return tasks, total, nil
}

func (r *TaskRepository) GetByIDAndUserID(id, userID uint) (*model.Task, error) {
    var task model.Task
    err := r.db.Where("id = ? AND user_id = ?", id, userID).First(&task).Error
    if errors.Is(err, gorm.ErrRecordNotFound) {
        return nil, ErrTaskNotFound
    }
    if err != nil {
        return nil, err
    }
    return &task, nil
}
```

注意：详情查询也必须带 `user_id`。

## 三、Service 查询

```go
func (s *TaskService) List(userID uint, page, pageSize int, status string) ([]model.Task, int64, error) {
    if page <= 0 {
        page = 1
    }
    if pageSize <= 0 {
        pageSize = 10
    }
    if pageSize > 100 {
        pageSize = 100
    }
    if status != "" && !isValidTaskStatus(status) {
        return nil, 0, ErrInvalidTaskStatus
    }

    return s.repo.ListByUserID(userID, page, pageSize, status)
}

func (s *TaskService) Get(userID, taskID uint) (*model.Task, error) {
    return s.repo.GetByIDAndUserID(taskID, userID)
}
```

分页默认值和上限放在 service。

## 四、Handler 查询参数

```go
type ListTasksRequest struct {
    Page     int    `form:"page" binding:"omitempty,gte=1"`
    PageSize int    `form:"page_size" binding:"omitempty,gte=1,lte=100"`
    Status   string `form:"status" binding:"omitempty,oneof=todo doing done"`
}

func (h *TaskHandler) List(c *gin.Context) {
    userID := c.GetUint("user_id")

    var req ListTasksRequest
    if err := c.ShouldBindQuery(&req); err != nil {
        response.Fail(c, http.StatusBadRequest, response.CodeInvalidRequest, "invalid query")
        return
    }

    tasks, total, err := h.service.List(userID, req.Page, req.PageSize, req.Status)
    if errors.Is(err, service.ErrInvalidTaskStatus) {
        response.Fail(c, http.StatusBadRequest, response.CodeInvalidRequest, "invalid task status")
        return
    }
    if err != nil {
        response.Fail(c, http.StatusInternalServerError, response.CodeInternalError, "internal server error")
        return
    }

    if req.Page == 0 {
        req.Page = 1
    }
    if req.PageSize == 0 {
        req.PageSize = 10
    }

    response.OK(c, response.PageResult{
        Items:    tasks,
        Total:    total,
        Page:     req.Page,
        PageSize: req.PageSize,
    })
}
```

## 五、Handler 详情

```go
func (h *TaskHandler) Get(c *gin.Context) {
    userID := c.GetUint("user_id")

    id, err := strconv.ParseUint(c.Param("id"), 10, 64)
    if err != nil {
        response.Fail(c, http.StatusBadRequest, response.CodeInvalidRequest, "invalid task id")
        return
    }

    task, err := h.service.Get(userID, uint(id))
    if errors.Is(err, repository.ErrTaskNotFound) {
        response.Fail(c, http.StatusNotFound, response.CodeNotFound, "task not found")
        return
    }
    if err != nil {
        response.Fail(c, http.StatusInternalServerError, response.CodeInternalError, "internal server error")
        return
    }

    response.OK(c, task)
}
```
