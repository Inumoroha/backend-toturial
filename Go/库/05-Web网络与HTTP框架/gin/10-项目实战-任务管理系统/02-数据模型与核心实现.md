## 三、Task Repository

`internal/repository/task_repository.go`

```go
package repository

import (
    "errors"

    "gin-learning/internal/model"

    "gorm.io/gorm"
)

var ErrTaskNotFound = errors.New("task not found")

type TaskRepository struct {
    db *gorm.DB
}

func NewTaskRepository(db *gorm.DB) *TaskRepository {
    return &TaskRepository{db: db}
}

func (r *TaskRepository) Create(task *model.Task) error {
    return r.db.Create(task).Error
}

func (r *TaskRepository) GetByIDAndUserID(id, userID uint) (*model.Task, error) {
    var task model.Task
    err := r.db.Where("id = ? AND user_id = ?", id, userID).First(&task).Error
    if errors.Is(err, gorm.ErrRecordNotFound) {
        return nil, ErrTaskNotFound
    }
    if err != nil {
        return nil, err
    }
    return &task, nil
}

func (r *TaskRepository) ListByUserID(userID uint, page, pageSize int, status string) ([]model.Task, int64, error) {
    var tasks []model.Task
    var total int64

    query := r.db.Model(&model.Task{}).Where("user_id = ?", userID)
    if status != "" {
        query = query.Where("status = ?", status)
    }

    if err := query.Count(&total).Error; err != nil {
        return nil, 0, err
    }

    offset := (page - 1) * pageSize
    err := query.Order("created_at DESC").Offset(offset).Limit(pageSize).Find(&tasks).Error
    if err != nil {
        return nil, 0, err
    }

    return tasks, total, nil
}

func (r *TaskRepository) Update(task *model.Task) error {
    return r.db.Save(task).Error
}

func (r *TaskRepository) Delete(id, userID uint) error {
    result := r.db.Where("id = ? AND user_id = ?", id, userID).Delete(&model.Task{})
    if result.Error != nil {
        return result.Error
    }
    if result.RowsAffected == 0 {
        return ErrTaskNotFound
    }
    return nil
}
```

注意所有查询都带 `user_id`。这是权限控制的一部分，不能只在前端做限制。

## 四、Task Service

`internal/service/task_service.go`

```go
package service

import (
    "errors"

    "gin-learning/internal/model"
    "gin-learning/internal/repository"
)

var ErrInvalidTaskStatus = errors.New("invalid task status")

type TaskService struct {
    repo *repository.TaskRepository
}

func NewTaskService(repo *repository.TaskRepository) *TaskService {
    return &TaskService{repo: repo}
}

type CreateTaskInput struct {
    UserID      uint
    Title       string
    Description string
    Status      string
}

func (s *TaskService) Create(input CreateTaskInput) (*model.Task, error) {
    status := input.Status
    if status == "" {
        status = "todo"
    }
    if !isValidTaskStatus(status) {
        return nil, ErrInvalidTaskStatus
    }

    task := &model.Task{
        UserID:      input.UserID,
        Title:       input.Title,
        Description: input.Description,
        Status:      status,
    }

    if err := s.repo.Create(task); err != nil {
        return nil, err
    }
    return task, nil
}

func (s *TaskService) Get(userID, taskID uint) (*model.Task, error) {
    return s.repo.GetByIDAndUserID(taskID, userID)
}

func (s *TaskService) List(userID uint, page, pageSize int, status string) ([]model.Task, int64, error) {
    if page <= 0 {
        page = 1
    }
    if pageSize <= 0 {
        pageSize = 10
    }
    if pageSize > 100 {
        pageSize = 100
    }
    if status != "" && !isValidTaskStatus(status) {
        return nil, 0, ErrInvalidTaskStatus
    }
    return s.repo.ListByUserID(userID, page, pageSize, status)
}

func isValidTaskStatus(status string) bool {
    switch status {
    case "todo", "doing", "done":
        return true
    default:
        return false
    }
}
```

业务规则放在 service：

- 默认状态。
- 状态是否合法。
- 分页默认值。
- 分页上限。

## 五、Task Handler

`internal/handler/task_handler.go`

```go
package handler

import (
    "errors"
    "net/http"
    "strconv"

    "gin-learning/internal/repository"
    "gin-learning/internal/response"
    "gin-learning/internal/service"

    "github.com/gin-gonic/gin"
)

type TaskHandler struct {
    service *service.TaskService
}

func NewTaskHandler(service *service.TaskService) *TaskHandler {
    return &TaskHandler{service: service}
}

type CreateTaskRequest struct {
    Title       string `json:"title" binding:"required,min=1,max=128"`
    Description string `json:"description" binding:"omitempty,max=2000"`
    Status      string `json:"status" binding:"omitempty,oneof=todo doing done"`
}

func (h *TaskHandler) Create(c *gin.Context) {
    userID := c.GetUint("user_id")

    var req CreateTaskRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.Fail(c, http.StatusBadRequest, 40001, "invalid request")
        return
    }

    task, err := h.service.Create(service.CreateTaskInput{
        UserID:      userID,
        Title:       req.Title,
        Description: req.Description,
        Status:      req.Status,
    })
    if errors.Is(err, service.ErrInvalidTaskStatus) {
        response.Fail(c, http.StatusBadRequest, 40001, "invalid task status")
        return
    }
    if err != nil {
        response.Fail(c, http.StatusInternalServerError, 50001, "internal server error")
        return
    }

    response.Created(c, task)
}

func (h *TaskHandler) Get(c *gin.Context) {
    userID := c.GetUint("user_id")

    id, err := strconv.ParseUint(c.Param("id"), 10, 64)
    if err != nil {
        response.Fail(c, http.StatusBadRequest, 40001, "invalid task id")
        return
    }

    task, err := h.service.Get(userID, uint(id))
    if errors.Is(err, repository.ErrTaskNotFound) {
        response.Fail(c, http.StatusNotFound, 40401, "task not found")
        return
    }
    if err != nil {
        response.Fail(c, http.StatusInternalServerError, 50001, "internal server error")
        return
    }

    response.OK(c, task)
}
```

## 七、实现重点

### 1. 用户只能访问自己的任务

不要这样写：

```go
db.First(&task, id)
```

应该这样：

```go
db.Where("id = ? AND user_id = ?", id, userID).First(&task)
```

这是最核心的安全边界。

### 2. 分页参数要有默认值和上限

否则用户可以传：

```text
page_size=100000
```

导致数据库压力过大。

### 3. 状态字段要限制枚举

不要允许任意字符串进入数据库。
